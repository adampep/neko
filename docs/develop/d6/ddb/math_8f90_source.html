407  subroutine vcross(u1, u2, u3, v1, v2, v3, w1, w2, w3, n)
408  integer, intent(in) :: n
409  real(kind=rp), dimension(n), intent(in) :: v1, v2, v3
410  real(kind=rp), dimension(n), intent(in) :: w1, w2, w3
411  real(kind=rp), dimension(n), intent(out) :: u1, u2, u3
412  integer :: i
413 
414  do i = 1, n
415  u1(i) = v2(i)*w3(i) - v3(i)*w2(i)
416  u2(i) = v3(i)*w1(i) - v1(i)*w3(i)
417  u3(i) = v1(i)*w2(i) - v2(i)*w1(i)
418  end do
419 
420  end subroutine vcross
